% text completion (0 - 100) outerColor innerColor textColor
\newcommand\checkBoxDrawItem[5] {

    \begin{scope}
        \clip [
            rounded corners = \checkBoxRadius,
        ]
        (tmp) rectangle ++(\checkBoxSide, -\checkBoxSide)
        ;

        \fill [
            #4,
        ]
        (tmp) rectangle ++(#2 / 100 * \checkBoxSide, -\checkBoxSide)
        ;
    \end{scope}

    \draw [
        very thick,
        #3,
        rounded corners = \checkBoxRadius,
    ]
    (tmp) rectangle ++(\checkBoxSide, -\checkBoxSide)
    ;

    \draw
    (tmp) ++(\checkBoxSide + \checkBoxPad, 0) ++(0, - \checkBoxSide / 2 + \textBlockHeight / 2 - 1.0pt)
    coordinate (text)
    node (text) [
        inner sep = 0pt,
        outer sep = 0pt,
        anchor = north west,
    ] {
        \begin{minipage} {\checkBoxTextWidth pt}
            \color{#5}
            #1
        \end{minipage}
    }
    ;

    \draw
    (tmp |- text.south) ++(0, -\checkBoxItemPad)
    coordinate (tmp)
    ;
}
