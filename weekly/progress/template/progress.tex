% <Date> <Holiday 1/0> <Work>
\newcommand\showDayWork[3] {
    \ifthenelse {#2 = 1} {
        \colorlet{colorTblDateColor}{colorSunLight}
        \colorlet{colorTblWorkColor}{white!20!colorSunLight}
    }{
        \colorlet{colorTblDateColor}{colorNonSun}
        \colorlet{colorTblWorkColor}{white!30!colorNonSun}
    }
    \begin{tikzpicture}
        \filldraw [
            colorTblDateColor,
            fill = colorTblDateColor,
            very thick,
            rounded corners = \tblRadius,
        ]
        (0, 0) ++(\tblDateCellWidth / 2, \tblCellHeight / 2) coordinate (dateCenter)
        (0, 0) rectangle ++(\tblDateCellWidth, \tblCellHeight) ++(\tblSpace, 0) coordinate (tmp)
        ;

        \filldraw [
            colorTblWorkColor,
            fill = colorTblWorkColor,
            very thick,
            rounded corners = \tblRadius,
        ]
        (tmp) rectangle ++(\tblWorkCellWidth, -\tblCellHeight)
        (tmp) ++(\tblWorkCellWidth / 2, -\tblCellHeight / 2) coordinate (workCenter)
        ;

        \draw [
        ]
        (dateCenter)
        node [anchor = center] {\texttt{#1}}
        ;

        \draw [
        ]
        (workCenter)
        node [anchor = center] {
            \pgfmathparse{\tblWorkCellWidth - \tblCellPad}
            \begin{minipage} {\pgfmathresult pt}
                \texttt{#3}
            \end{minipage}
        }
        ;

    \end{tikzpicture}
}
