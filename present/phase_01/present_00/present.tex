\begin{frame} {Introduction}

    {
        \begin{minipage} {0.415\textwidth}
            \begin{center}
                \includeObjImage{ext_right.png}{5}{4.5cm}{0.5cm}
            \end{center}
        \end{minipage}
        \begin{minipage} {0.55\textwidth}
            \textbf{Seed Incubation Plant} aims to optimize the germination and
            growth process of seeds by providing a controlled and
            monitored environment.
            While the \textbf{Environment Monitoring and Control Unit} (EMCU) part of the
            incubator takes care of environmental factors such as temperature,
            humidity, lighting, $\mbox{O}_2$, etc, the
            \textbf{Growth Monitoring Unit} (GMU) closely monitors and collects growth related
            information from each of the germinating saplings.
            By the help of \textbf{SINC}, a companion app, the status of the incubator
            can be closely watched.

        \end{minipage}
    }

\end{frame}
